238 /**
239 * This will return a value lower than the input, and it is used to slow
240 * down the motors during stop mode
241 *
242 * @param inputVelocity between -1 and 1 (double)
243 * @return the new value (lower)
244 */
245 public double slowDown(double inputVelocity){
246 // input is between -1 and 1
247 double newVelocity;
248 if (Math.abs(inputVelocity) > Constants.SLOW_DOWN_CUTOFF){
249 // velocity still needs to be reduced (magnatude is above cutoff)
250 newVelocity = inputVelocity / Constants.SLOW_DOWN_FACTOR;
251 } else {
252 // input has reached cutoff, now returning 0 speed
253 return 0;
254 }
255 return newVelocity;
256 }

392 * PID Tuning Mode: Increments the selected PID constant
393 */
394 public void incrementPIDConstant() {
395 if(increasingPIDConstant) {
396 PIDConstants[currentPIDConstant] += PIDIncrements[currentPIDConstant];
397 } else {
398 PIDConstants[currentPIDConstant] -= PIDIncrements[currentPIDConstant];
399 }
400 // make sure constants are in [0, constantMax]
401 PIDConstants[currentPIDConstant] = Math.min(PIDConstants[currentPIDConstant], PIDMaximums[currentPIDConstant]);
402 PIDConstants[currentPIDConstant] = Math.max(PIDConstants[currentPIDConstant], 0);
403
404 // print PID constants
405 System.out.println("kP: " + PIDConstants[0] + ", kI: " + PIDConstants[1] + ", kD: " + PIDConstants[2]);
406
407 // set PID constant in the PID controllers
408 if(currentPIDConstant == 0) {
409 leftPID.setP(PIDConstants[0]);
410 rightPID.setP(PIDConstants[0]);
411 } else if (currentPIDConstant == 1) {
412 leftPID.setI(PIDConstants[1]);
413 rightPID.setI(PIDConstants[1]);
414 } else if (currentPIDConstant == 2) {
415 leftPID.setD(PIDConstants[2]);
416 rightPID.setD(PIDConstants[2]);
417 }
418 }
