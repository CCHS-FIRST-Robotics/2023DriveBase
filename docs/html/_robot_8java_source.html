98 /** This function is called periodically during operator control. */
99 @Override
100 public void teleopPeriodic() {
101 // TODO: put the driving loop here
102 // TODO: check if buttons are pressed and comment controller mapping
103 if (xboxController.getRightBumperPressed()) driveBase.increaseSpeedBracket();
104 if (xboxController.getLeftBumperPressed()) driveBase.decreaseSpeedBracket();
105
106 // switch between modes (for DPad, 0 is up, and angles go clockwise, so 90 is right)
107 // up
108 if (xboxController.getPOV() == 0) driveBase.turnOnDefaultMode();
109 // right
110 if (xboxController.getPOV() == 90) driveBase.turnOnStopMode();
111 // left
112 if (xboxController.getPOV() == 270) driveBase.turnOnDebugMode();
113 // down
114 if (xboxController.getPOV() == 180) driveBase.turnONPIDTurningMode();
115
116
117 // Debug controls
118 if (xboxController.getRightStickButtonPressed()) driveBase.cycleWheelDebugMode();
119
120 // actions of the four button presses
121 if (xboxController.getAButtonPressed()) {
122 driveBase.AButtonPressed();
123 }
124 if (xboxController.getBButtonPressed()) {
125 driveBase.BButtonPressed();
126 }
127 if (xboxController.getXButtonPressed()) {
128 driveBase.XButtonPressed();
129 }
130 if (xboxController.getYButtonPressed()) {
131 driveBase.YButtonPressed();
132 }
133
134 // get analog input from xbox controller
135 double leftAnalogX = xboxController.getLeftX();
136 double leftAnalogY = xboxController.getLeftY();
137 double rightAnalogX = xboxController.getRightX();
138 double rightAnalogY = xboxController.getRightY();
139
140 // process input (determine wheelspeeds)
141 driveBase.drive(leftAnalogX, leftAnalogY, rightAnalogX, rightAnalogY);
142 }
