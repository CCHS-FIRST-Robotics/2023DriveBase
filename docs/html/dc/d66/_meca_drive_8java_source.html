45 /**
46 * drive the robot using controller inputs
47 *
48 * Left analog stick controls translation
49 * Right analog stick controls rotation (move it right -> rotate clockwise, move it left -> rotate counterclockwise)
50 *
51 * positive is right/up
52 *
53 * @param leftAnalogX the x position of the left analog stick; range: [-1, 1]
54 * @param leftAnalogY the y position of the left analog stick; range: [-1, 1]
55 * @param rightAnalogX the x position of the right analog stick; range: [-1, 1]
56 * @param rightAnalogY the y position of the right analog stick; range: [-1, 1]
57 *
58 * Debug mode:
59 *
60 * When you press the left stick down (left stick button) the robot enters
61 * debug mode (bool debugMode), and the robot will only move power one wheel
62 * at a time. Toggle through these wheels by pressing the left stick button
63 * more (FL -> FR -> BL -> BR) then, after BR, it will leave debug mode and
64 * go back into all wheel drive
65 *
66 */
67 public void drive(double leftAnalogX, double leftAnalogY,
68 double rightAnalogX, double rightAnalogY) {
69
70 // left analog stick controls translation
71 // right analog stick controls rotation
72
73 leftAnalogY *= verticalSpeedMultiplier;
74 leftAnalogX *= horizontalSPeedMultiplier;
75
76 // arrays for wheel speeds (percents)
77 // 1st is front left, 2nd is front right, 3rd is back left, 4th is back right
78 double[] verticalSpeeds = {leftAnalogY, leftAnalogY,
79 leftAnalogY, leftAnalogY};
80
81 // negatives due to wheels going in opposite directions during left or right translation
82 double[] horizontalSpeeds = {-1 * leftAnalogX, leftAnalogX,
83 leftAnalogX, -1 * leftAnalogX};
84
85 // left and right wheels should go different directions to rotate the robot
86 double[] rotationSpeeds = {-1 * rightAnalogX, rightAnalogX,
87 -1 * rightAnalogX, rightAnalogX};
88
89 // so these could exceed 1 (not good; we cannot run the motors at over 100%)
90 // we will use the maximum speed to scale all the other speeds to something below 1
91 double[] combinedSpeeds = new double[4];
92 for (int i = 0; i < 4; i ++)
93 combinedSpeeds[i] = verticalSpeeds[i] + horizontalSpeeds[i] + rotationSpeeds[i];
94
95 // find the max of the above speeds so we can check if it is above 1
96 double maxSpeed = Integer.MIN_VALUE;
97 for (int i = 0; i < 4; i++)
98 if (Math.abs(combinedSpeeds[i]) > maxSpeed) maxSpeed = Math.abs(combinedSpeeds[i]);
99
100 maxSpeed = Math.max(1, maxSpeed); // if it is under 1, we can basically ignore it
101
102 for (int i = 0; i < 4; i++) {
103 // scale the speeds
104 combinedSpeeds[i] = (1 / maxSpeed) * combinedSpeeds[i];
105
106 // we also need to scale the speeds by the speed multiplier
107 combinedSpeeds[i] = combinedSpeeds[i] * speedMultiplier;
108 }
109
110 // set the motor speeds (normal)
111 if(!debugMode) {
112 frontLeftMotor.set(ControlMode.PercentOutput, combinedSpeeds[0] * -1);
113 frontRightMotor.set(ControlMode.PercentOutput, combinedSpeeds[1]);
114 rearLeftMotor.set(ControlMode.PercentOutput, combinedSpeeds[2] * -1);
115 rearRightMotor.set(ControlMode.PercentOutput, combinedSpeeds[3]);
116 }
117
118 // Debug mode (toggle wheels with left stick button)
119 else {
120 switch (debugEnabledWheel){
121 case 0:
122 frontLeftMotor.set(ControlMode.PercentOutput, combinedSpeeds[0] * -1);
123 break;
124 case 1:
125 frontRightMotor.set(ControlMode.PercentOutput, combinedSpeeds[1]);
126 break;
127 case 2:
128 rearLeftMotor.set(ControlMode.PercentOutput, combinedSpeeds[2] * -1);
129 break;
130 case 3:
131 rearRightMotor.set(ControlMode.PercentOutput, combinedSpeeds[3]);
132 break;
133 }
134 }
135 }
