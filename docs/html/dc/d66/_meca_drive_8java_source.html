147 private double[] combineSpeeds(double leftAnalogX, double leftAnalogY,
148 double rightAnalogX, double rightAnalogY){
149
150 leftAnalogY *= verticalSpeedMultiplier;
151 leftAnalogX *= horizontalSPeedMultiplier;
152
153 // arrays for wheel speeds (percents)
154 // 1st is front left, 2nd is front right, 3rd is back left, 4th is back right
155 double[] verticalSpeeds = {leftAnalogY, leftAnalogY,
156 leftAnalogY, leftAnalogY};
157
158 // negatives due to wheels going in opposite directions during left or right translation
159 double[] horizontalSpeeds = {-1 * leftAnalogX, leftAnalogX,
160 leftAnalogX, -1 * leftAnalogX};
161
162 // left and right wheels should go different directions to rotate the robot
163 double[] rotationSpeeds = {-1 * rightAnalogX, rightAnalogX,
164 -1 * rightAnalogX, rightAnalogX};
165
166 // so these could exceed 1 (not good; we cannot run the motors at over 100%)
167 // we will use the maximum speed to scale all the other speeds to something below 1
168 for (int i = 0; i < 4; i ++)
169 combinedSpeeds[i] = verticalSpeeds[i] + horizontalSpeeds[i] + rotationSpeeds[i];
170
171 // find the max of the above speeds so we can check if it is above 1
172 double maxSpeed = Integer.MIN_VALUE;
173
174 for (int i = 0; i < 4; i++)
175 if (Math.abs(combinedSpeeds[i]) > maxSpeed) maxSpeed = Math.abs(combinedSpeeds[i]);
176
177 maxSpeed = Math.max(1, maxSpeed); // if it is under 1, we can basically ignore it
178
179 for (int i = 0; i < 4; i++) {
180 // scale the speeds
181 combinedSpeeds[i] = (1 / maxSpeed) * combinedSpeeds[i];
182
183 // we also need to scale the speeds by the speed multiplier
184 combinedSpeeds[i] = combinedSpeeds[i] * speedMultiplier;
185 }
186
187 return combinedSpeeds;
188 }
