150 /**
151 * Switches between different driving modes
152 * For the DPad, 0 is up, and angles go clockwise, so 90 is right
153 */
154 private void checkForModeSwitches() {
155 // up
156 if (xboxController.getPOV() == 0) driveBase.turnOnDefaultMode();
157 // right
158 if (xboxController.getPOV() == 90) driveBase.turnOnStopMode();
159 // left
160 if (xboxController.getPOV() == 270) driveBase.turnOnDebugMode();
161 // down
162 if (xboxController.getPOV() == 180) driveBase.turnONPIDTuningMode();
163 }
164
165 /**
166 * Checks for button presses and activates their functions
167 */
168 private void checkForButtonPresses() {
169 if (xboxController.getAButtonPressed()) {
170 driveBase.cycleMotor();
171 }
172 if (xboxController.getBButtonPressed()) {
173 driveBase.printActiveMotorDebugMode();
174 }
175 if (xboxController.getLeftBumperPressed()) {
176 driveBase.decreaseSpeedBracket();
177 }
178 if (xboxController.getRightBumperPressed()) {
179 driveBase.increaseSpeedBracket();
180 }
181 }
182

190 }
