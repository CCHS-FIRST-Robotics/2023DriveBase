78 private double linearizeVelocity(double velocity, double maxSpeed) {
79 // Speed is a scalar, (velocity is not scalar, negative matters) which means that maxSpeed has no direction, and thus a negative value in this case would imply a incorrect usage for maxSpeed.
80 if (maxSpeed <= 0){
81 throw new IllegalArgumentException("maxSpeed must be positive and greater than 0");
82 }
83
84 double linearizedVelocity; // from 1 to -1
85
86 if (velocity == 0) {
87 linearizedVelocity = 0;
88 }
89 else if (velocity < 0) {
90 // if velocity is in greater magnitude than maxSpeed, divide it by maxspeed so that it is in the range [0, 1]
91 linearizedVelocity = Math.max(-1, velocity / maxSpeed);
92 }
93 else {
94 linearizedVelocity = Math.min(1, velocity / maxSpeed);
95 }
